aztec_war_spawn_troops_effect = {
	if = {
		limit = {
			NOR = {
				has_global_flag = aztecs_acclimitised_to_horses
				has_global_flag = aztecs_employing_gunpowder
			}
		}
		create_random_soldier = {
			random_traits = yes
			dynasty = random
			religion = aztec
			culture = nahuatl
			female = no
			age = 30
			attributes = {
				martial = 12
			}
			trait = brilliant_strategist
		}
		new_character = {
			save_event_target_as = aztec_new_troop_commander
		}
		while = {
			limit = {
				check_variable = {
					which = aztec_troop_quantity
					value = 1
				}
			}
			
			capital_scope = {
				event_target:aztec_new_troop_commander = {
					random_list = {
						10 = {
							trigger = {
								check_variable = {
									which = aztec_troop_quality
									value = 1
								}
								NOT = {
									check_variable = {
										which = aztec_troop_quality
										value = 2
									}
								}
							}
							spawn_unit = {
								province = event_target:spawn_province
								home = event_target:spawn_province
								troops =
								{
									archers = { 168 168 }
									light_infantry = { 284 284 }
									heavy_infantry = { 0 0 }
									pikemen = { 0 0 }
								}
								disband_on_peace = yes
								merge = yes
							}
						}
						10 = {
							trigger = {
								check_variable = {
									which = aztec_troop_quality
									value = 2
								}
								NOT = {
									check_variable = {
										which = aztec_troop_quality
										value = 3
									}
								}
							}
							spawn_unit = {
								province = event_target:spawn_province
								home = event_target:spawn_province
								troops =
								{
									archers = { 168 168 }
									light_infantry = { 192 192 }
									heavy_infantry = { 140 140 }
									pikemen = { 20 20 }
								}
								disband_on_peace = yes
								merge = yes
							}
						}
						10 = {
							trigger = {
								check_variable = {
									which = aztec_troop_quality
									value = 3
								}
								#NOT = {
								#	check_variable = {
								#		which = aztec_troop_quality
								#		value = 4
								#	}
								#}
							}
							spawn_unit = {
								province = event_target:spawn_province
								home = event_target:spawn_province
								troops =
								{
									archers = { 168 168 }
									light_infantry = { 192 192 }
									heavy_infantry = { 160 160 }
									pikemen = { 140 140 }
								}
								disband_on_peace = yes
								merge = yes
							}
						}
					}
				}
			}
			
			change_variable = {	
				which = aztec_troop_quantity
				value = -1
			}
		}
	}
}
